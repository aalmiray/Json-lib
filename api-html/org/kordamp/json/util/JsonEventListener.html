18 package org.kordamp.json.util;
19 
20 import org.kordamp.json.JSONException;
21 
22 /**
23  * Defines the contract to handle JsonEvents when building an object or array.
24  *
25  * @author Andres Almiray
26  */
27 public interface JsonEventListener {
28 
29     /**
30      * Ttriggered when reaching the end of an array.
31      */
32     void onArrayEnd();
33 
34     /**
35      * triggered when the start of an array is encountered.
36      */
37     void onArrayStart();
38 
39     /**
40      * Triggered when an element has been added to the current array.
41      *
42      * @param index   the index where the element was added
43      * @param element the added element
44      */
45     void onElementAdded(int index, Object element);
46 
47     /**
48      * Triggered when an exception is thrown.
49      *
50      * @param jsone the thrown exception
51      */
52     void onError(JSONException jsone);
53 
54     /**
55      * triggered when reaching the end of an object.
56      */
57     void onObjectEnd();
58 
59     /**
60      * Triggered when the start of an object is encountered.
61      */
62     void onObjectStart();
63 
64     /**
65      * Triggered when a property is set on an object
66      *
67      * @param key         the name of the property
68      * @param value       the value of the property
69      * @param accumulated if the value has been accumulated over 'key'
70      */
71     void onPropertySet(String key, Object value, boolean accumulated);
72 
73     /**
74      * Triggered when a warning is encountered.
75      *
76      * @param warning the warning message
77      */
78     void onWarning(String warning);
79 }
