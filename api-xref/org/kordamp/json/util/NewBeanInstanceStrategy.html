18  package org.kordamp.json.util;
19  
20  import org.kordamp.json.JSONObject;
21  
22  import java.lang.reflect.Constructor;
23  import java.lang.reflect.InvocationTargetException;
24  
25  /**
26   * Base class for creating Bean instances.<br>
27   * <ul>
28   * <li>DEFAULT - calls Class.newInstance().</li>
29   * </ul>
30   *
31   * @author Andres Almiray
32   */
33  public abstract class NewBeanInstanceStrategy {
34      /**
35       * Calls Class.newInstance()
36       */
37      public static final NewBeanInstanceStrategy DEFAULT = new DefaultNewBeanInstanceStrategy();
38  
39      /**
40       * Creates a new instance.
41       *
42       * @param target the source class
43       * @param source additional properties that may be needed to create the
44       *               instance
45       */
46      public abstract Object newInstance(Class target, JSONObject source)
47          throws InstantiationException, IllegalAccessException, SecurityException,
48          NoSuchMethodException, InvocationTargetException;
49  
50      private static final class DefaultNewBeanInstanceStrategy extends NewBeanInstanceStrategy {
51          private static final Object[] EMPTY_ARGS = new Object[0];
52          private static final Class[] EMPTY_PARAM_TYPES = new Class[0];
53  
54          public Object newInstance(Class target, JSONObject source) throws InstantiationException,
55              IllegalAccessException, SecurityException, NoSuchMethodException,
56              InvocationTargetException {
57              if (target != null) {
58                  Constructor c = target.getDeclaredConstructor(EMPTY_PARAM_TYPES);
59                  c.setAccessible(true);
60                  try {
61                      return c.newInstance(EMPTY_ARGS);
62                  } catch (InstantiationException e) {
63                      // getCause() was added on jdk 1.4
64                      String cause = "";
65                      try {
66                          cause = e.getCause() != null ? "\n" + e.getCause().getMessage() : "";
67                      } catch (Throwable t) { /* ignore */ }
68                      throw new InstantiationException(
69                          "Instantiation of \"" + target + "\" failed. " +
70                              "It's probably because class is an interface, " +
71                              "abstract class, array class, primitive type or void." +
72                              cause);
73                  }
74              }
75              return null;
76          }
77      }
78  }
