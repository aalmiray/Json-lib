61  /**

68   * JSONObject json = JSONObject.fromObject("{\"name\":\"json\",\"bool\":true,\"int\":1}");

71   * <name type="string">json</name>
72   * <bool type="boolean">true</bool>

75   * </pre><pre>
76   * JSONArray json = JSONArray.fromObject("[1,2,3]");
77   * String xml = new XMLSerializer().write( json );

79   * <e type="number">1</e>
80   * <e type="number">2</e>

262      * @param prefix      namespace prefix
263      * @param uri         namespace uri

266     public void addNamespace(String prefix, String uri, String elementName) {
267         if (StringUtils.isBlank(uri)) {
268             return;
269         }
270         if (StringUtils.isBlank(prefix)) {
271             prefix = "";
272         }
273         if (StringUtils.isBlank(elementName)) {
274             rootNamespace.put(prefix.trim(), uri.trim());
275         } else {
276             Map nameSpaces = (Map) namespacesPerElement.get(elementName);
277             if (nameSpaces == null) {
278                 nameSpaces = new TreeMap();
279                 namespacesPerElement.put(elementName, nameSpaces);
280             }
281             nameSpaces.put(prefix, uri);
282         }
283     }

531     /**
532      * Creates a JSON value from a XML string.
533      *
534      * @param xml A well-formed xml document in a String
535      *
536      * @return a JSONNull, JSONObject or JSONArray
537      *
538      * @throws JSONException if the conversion from XML to JSON can't be made for
539      *                       I/O or format reasons.
540      */
541     public JSON read(String xml) {
542         JSON json = null;
543         try {
544             Document doc = new Builder().build(new StringReader(xml));
545             Element root = doc.getRootElement();
546             if (isNullObject(root)) {
547                 return JSONNull.getInstance();
548             }
549             String defaultType = getType(root, JSONTypes.STRING);
550             if (isArray(root, true)) {
551                 json = processArrayElement(root, defaultType);
552                 if (forceTopLevelObject) {
553                     String key = removeNamespacePrefix(root.getQualifiedName());
554                     json = new JSONObject().element(key, json);
555                 }
556             } else {
557                 json = processObjectElement(root, defaultType);
558                 if (forceTopLevelObject) {
559                     String key = removeNamespacePrefix(root.getQualifiedName());
560                     json = new JSONObject().element(key, json);
561                 }
562             }
563         } catch (JSONException jsone) {
564             throw jsone;
565         } catch (Exception e) {
566             throw new JSONException(e);
567         }
568         return json;
569     }

572      * Creates a JSON value from a File.
573      *
574      * @param file
575      *
576      * @return a JSONNull, JSONObject or JSONArray
577      *
578      * @throws JSONException if the conversion from XML to JSON can't be made for
579      *                       I/O or format reasons.
580      */
581     public JSON readFromFile(File file) {
582         if (file == null) {
583             throw new JSONException("File is null");
584         }
585         if (!file.canRead()) {
586             throw new JSONException("Can't read input file");
587         }
588         if (file.isDirectory()) {
589             throw new JSONException("File is a directory");
590         }
591         try {
592             return readFromStream(new FileInputStream(file));
593         } catch (IOException ioe) {
594             throw new JSONException(ioe);
595         }
